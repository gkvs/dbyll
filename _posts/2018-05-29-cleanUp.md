How to remove all records in table?
Here is the solution:

{% highlight java %}
public void cleanUp(String table) {

	Iterator<Record> i = GKVS.Client.scan(table)
			.includeKey(true)
			.includeValue(false)
			.sync();

	while(i.hasNext()) {

		try {
			GKVS.Client.remove(i.next().key().get()).sync();
		}
		catch(GKVSException e) {
			e.printStackTrace();
		}

	}

}
{% endhighlight %}
